// frontend/src/hooks/useAuth.ts — EXPORTS

export interface AuthState {
  status: 'loading' | 'unauthenticated' | 'authenticated'
  idToken: string | null     // Cognito ID token — pass this to useWebSocket as JWT
  email: string | null
  error: string | null
}

export interface UseAuthReturn extends AuthState {
  signIn: (email: string, password: string) => Promise<void>
  signUp: (email: string, password: string) => Promise<void>
  signOut: () => void
}

export function useAuth(): UseAuthReturn

    <!-- Full API reference — executor needs no external research. -->

    ```typescript
    import {
      CognitoUserPool,
      CognitoUser,
      AuthenticationDetails,
      CognitoUserAttribute,
      CognitoUserSession,
    } from 'amazon-cognito-identity-js'

    // Instantiate pool (singleton or per-hook — per-hook is fine for tests)
    const userPool = new CognitoUserPool({
      UserPoolId: import.meta.env.VITE_COGNITO_USER_POOL_ID,
      ClientId:   import.meta.env.VITE_COGNITO_CLIENT_ID,
    })

    // ── SIGN IN ──────────────────────────────────────────────────────────────
    const cognitoUser = new CognitoUser({ Username: email, Pool: userPool })
    const authDetails = new AuthenticationDetails({ Username: email, Password: password })
    cognitoUser.authenticateUser(authDetails, {
      onSuccess: (session: CognitoUserSession) => {
        const idToken      = session.getIdToken().getJwtToken()   // ← send to gateway
        const refreshToken = session.getRefreshToken().getToken()
        localStorage.setItem('auth_id_token', idToken)
        localStorage.setItem('auth_refresh_token', refreshToken)
        localStorage.setItem('auth_email', email)
        // → set status='authenticated', idToken, email, error=null
      },
      onFailure: (err: Error) => {
        // → set error=err.message, status='unauthenticated'
      },
      newPasswordRequired: (_userAttrs: Record<string, unknown>) => {
        // → set error='Password change required — contact admin', status='unauthenticated'
      },
    })

    // ── SIGN UP ──────────────────────────────────────────────────────────────
    userPool.signUp(
      email,
      password,
      [new CognitoUserAttribute({ Name: 'email', Value: email })],
      [],
      (err, _result) => {
        if (err) { /* → set error=err.message */ return }
        // Auto-sign-in after successful signup by calling signIn(email, password)
      }
    )

    // ── SIGN OUT ─────────────────────────────────────────────────────────────
    userPool.getCurrentUser()?.signOut()
    localStorage.removeItem('auth_id_token')
    localStorage.removeItem('auth_refresh_token')
    localStorage.removeItem('auth_email')
    // → set status='unauthenticated', idToken=null, email=null, error=null

    // ── SESSION RESTORE (on mount) ────────────────────────────────────────────
    // Runs in useEffect on mount — sets status='loading' then resolves.
    const storedUser = userPool.getCurrentUser()
    if (!storedUser) {
      // → set status='unauthenticated'
    } else {
      storedUser.getSession((err: Error | null, session: CognitoUserSession | null) => {
        if (err || !session || !session.isValid()) {
          // → set status='unauthenticated'
        } else {
          const idToken = session.getIdToken().getJwtToken()
          const email   = storedUser.getUsername()
          // → set status='authenticated', idToken, email, error=null
        }
      })
    }
    ```
