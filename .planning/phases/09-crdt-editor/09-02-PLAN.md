```typescript
export interface UseCRDTOptions {
  sendMessage: (msg: Record<string, unknown>) => void;
  onMessage: (handler: (msg: GatewayMessage) => void) => () => void;
  currentChannel: string;
  connectionState: ConnectionState;
}

export interface UseCRDTReturn {
  content: string;                    // Current Y.Text content as plain string (reactive)
  applyLocalEdit: (newText: string) => void;  // Called by editor on user input
}

export function useCRDT(options: UseCRDTOptions): UseCRDTReturn;
```

```typescript
// Feature hook message handler registry
const featureHandlers = useRef<Array<(msg: GatewayMessage) => void>>([]);

// Stable onMessage registrar passed to feature hooks
const onMessage = (handler: (msg: GatewayMessage) => void) => {
  featureHandlers.current.push(handler);
  return () => {
    featureHandlers.current = featureHandlers.current.filter((h) => h !== handler);
  };
};

// Existing feature hooks (reference for placement):
```
