```typescript
// Get character offset at click/keyup position:
function getCharOffset(container: HTMLElement): number {
  const sel = window.getSelection();
  if (!sel || sel.rangeCount === 0) return 0;
  const range = document.createRange();
  range.setStart(container, 0);
  range.setEnd(sel.anchorNode!, sel.anchorOffset);
  return range.toString().length;
}

// Get selection data:
function getSelectionData(container: HTMLElement): { start: number; end: number; text: string } | null {
  const sel = window.getSelection();
  if (!sel || sel.isCollapsed) return null;
  const range = sel.getRangeAt(0);
  const preStart = document.createRange();
  preStart.setStart(container, 0);
  preStart.setEnd(range.startContainer, range.startOffset);
  const start = preStart.toString().length;
  return { start, end: start + range.toString().length, text: range.toString() };
}
```
